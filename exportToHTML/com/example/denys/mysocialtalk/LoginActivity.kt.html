3    import android.content.Intent

10    
11   class LoginActivity : AppCompatActivity(){ 
12    
13       companion object { 
14           val TAG = "LoginActivity" 
15       } 
16    
17       override fun onCreate(savedInstanceState: Bundle?) { 
18           super.onCreate(savedInstanceState) 
19    
20           setContentView(R.layout.activity_login) 
21    
22           login_button_login.setOnClickListener{ 
23               performLogin() 
24           } 
25    
26           dont_have_account_textView_login.setOnClickListener(){ 
27    
28               Log.d(TAG,"Try to show log activity") 
29    
30               val intent = Intent(this, RegisterActivity::class.java) 
31               startActivity(intent) 
32           } 
33    
34       } 
35    
36       private fun performLogin(){ 
37           val email = email_editText_login.text.toString() 
38           val password = password_editText_login.text.toString() 
39    
40           if (email.isEmpty()) { 
41               Toast.makeText(this, "Please enter the email",Toast.LENGTH_SHORT).show() 
42               return 
43           } 
44    
45           if (password.isEmpty()) { 
46               Toast.makeText(this, "Please enter the password", Toast.LENGTH_SHORT).show() 
47               return 
48           } 
49    
50           FirebaseAuth.getInstance().signInWithEmailAndPassword(email, password) 
51                   .addOnCompleteListener { 
52                       val intent = Intent(this, LatestMessagesActivity::class.java) 
53                       startActivity(intent) 
54                   } 
55                   .addOnFailureListener{ 
56                       Toast.makeText(this, "Failed to login, check your email and password", 
57                               Toast.LENGTH_SHORT).show() 
58                       return@addOnFailureListener 
59                   } 
60    
61       } 
62   }
